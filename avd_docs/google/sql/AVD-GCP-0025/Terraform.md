
Enable checkpoints logging.

```hcl
 resource "google_sql_database_instance" "db" {
 	name             = "db"
 	database_version = "POSTGRES_12"
 	region           = "us-central1"
 	settings {
 		database_flags {
 			name  = "log_checkpoints"
 			value = "on"
 		}
 	}
 }
 			
```
